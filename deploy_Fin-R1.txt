# 查看磁盘剩余容量是否足够下载权重
df -h /home/data
sh download_Fin-R1.sh

# docker images下载Dir也设置到这目录
# 下载vllm驱动，我这里是A800，适配v0.4.0
nohup docker pull --disable-content-trust vllm/vllm-openai:v0.4.0 > docker_pull.log 2>&1 &

# 启动Fin-R1 API Server
# nvidia-smi 查看到2号GPU目前空闲，gpus device=2
# lsof -i:8010未占用，port 8010
docker run -d \
  --name fin-r1-vllm \
  --gpus device=2 \
  --network host \
  --shm-size=16g \
  -v /home/data/Fin-R1:/models/Fin-R1 \
  vllm/vllm-openai:v0.4.0 \
  --model /models/Fin-R1 \
  --gpu-memory-utilization 0.8 \
  --max-num-batched-tokens 32768 \
  --max-model-len 32768 \
  --port 8010 \
  --host 0.0.0.0

# 测试API
curl http://192.168.0.198:8010/v1/chat/completions \
  -H "Content-Type: application/json" \
  -d '{
    "model": "/models/Fin-R1",
    "messages": [{"role": "user", "content": "分析与两会主旨密切相关的受益股"}]
  }'
  
# 关闭Fin-R1
docker stop fin-r1-vllm && docker rm fin-r1-vllm

# 下载chat-ui通用Web UI
nohup docker pull --disable-content-trust yidadaa/chatgpt-next-web:latest > chatui_pull.log 2>&1 &

# 启动chatgpt-next-web对接Fin-R1 API
docker run -d \
  --name fin-r1-webui \
  -p 8011:3000 \
  -e BASE_URL=http://172.17.0.1:8010 \
  -e API_KEY="none" \
  -e HOST="0.0.0.0" \
  -e PORT=3000 \
  -e DEFAULT_MODEL="/models/Fin-R1" \
  -e DISABLE_OPENAI_MODEL_LIST=true \
  -e CUSTOM_MODELS="/models/Fin-R1" \
  yidadaa/chatgpt-next-web:latest
